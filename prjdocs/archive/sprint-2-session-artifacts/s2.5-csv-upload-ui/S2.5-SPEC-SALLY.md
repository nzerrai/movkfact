```jsx
// UploadZone.jsx
import { useState, useRef } from 'react';
import { Box, Button, Card, Typography } from '@mui/material';

const UploadZone = ({ onFileSelected, onFileDropped }) => {
  const [dragActive, setDragActive] = useState(false);
  const fileInput = useRef(null);

  const handleDrag = (e) => {
    e.preventDefault();
    setDragActive(e.type === "dragenter" || e.type === "dragover");
  };

  const handleDrop = (e) => {
    e.preventDefault();
    setDragActive(false);
    const file = e.dataTransfer.files[0];
    
    if (file.type === 'text/csv' || file.name.endsWith('.csv')) {
      onFileDropped(file);
    } else {
      // Show error: only CSV files
    }
  };

  return (
    <Card
      onDragEnter={handleDrag}
      onDragLeave={handleDrag}
      onDragOver={handleDrag}
      onDrop={handleDrop}
      sx={{
        padding: 4,
        textAlign: 'center',
        border: '2px dashed',
        borderColor: dragActive ? 'primary.main' : 'divider',
        backgroundColor: dragActive ? 'action.hover' : 'background.paper',
        cursor: 'pointer',
        transition: 'all 0.3s'
      }}
    >
      <Typography variant="h6">📁 CSV Upload & Preview</Typography>
      <Typography sx={{ my: 2 }}>
        Drag CSV file here or click below
      </Typography>
      <Button
        variant="contained"
        onClick={() => fileInput.current?.click()}
      >
        📂 Select File
      </Button>
      <input
        ref={fileInput}
        type="file"
        accept=".csv"
        hidden
        onChange={(e) => onFileSelected(e.target.files[0])}
      />
      <Typography variant="caption" sx={{ mt: 2, display: 'block', color: 'text.secondary' }}>
        CSV files only • Maximum 10 MB
      </Typography>
    </Card>
  );
};

export default UploadZone;
```

```jsx
// TypeDetectionResults.jsx
import { Box, Button, Card, CardContent, Chip, TextField, Typography } from '@mui/material';

const TypeDetectionResults = ({
  results,
  typeOverrides,
  onTypeOverride,
  onConfirm,
  onCancel
}) => {
  const getConfidenceColor = (confidence) => {
    if (confidence >= 0.8) return 'success';
    if (confidence >= 0.6) return 'warning';
    return 'error';
  };

  const getConfidenceIcon = (confidence) => {
    if (confidence >= 0.8) return '🟢';
    if (confidence >= 0.6) return '🟡';
    return '🔴';
  };

  const allColumnTypes = [
    // Personal (6)
    'PERSONAL_ID', 'FULL_NAME', 'EMAIL', 'PHONE', 'ADDRESS', 'SSN',
    // Financial (3)
    'AMOUNT', 'BANK_ACCOUNT', 'CURRENCY',
    // Temporal (4)
    'DATE', 'TIME', 'DATETIME', 'DURATION'
  ];

  return (
    <Card>
      <CardContent>
        <Typography variant="h6">📊 Column Type Detection</Typography>

        {results.map((column) => {
          const selectedType = typeOverrides[column.name] || column.type;
          const selectedConfidence = typeOverrides[column.name] === column.type 
            ? column.confidence 
            : 0.5; // Lower confidence if manually overridden

          return (
            <Box
              key={column.name}
              sx={{
                display: 'flex',
                alignItems: 'center',
                gap: 2,
                padding: 2,
                borderBottom: '1px solid #eee',
                '&:hover': { backgroundColor: '#f9f9f9' }
              }}
            >
              {/* Column Name */}
              <Typography sx={{ minWidth: 120, fontWeight: 'bold' }}>
                {column.name}
              </Typography>

              {/* Type Dropdown */}
              <TextField
                select
                value={selectedType}
                onChange={(e) => onTypeOverride(column.name, e.target.value)}
                SelectProps={{ native: true }}
                size="small"
                sx={{ minWidth: 150 }}
              >
                {allColumnTypes.map((type) => (
                  <option key={type} value={type}>
                    {type}
                  </option>
                ))}
              </TextField>

              {/* Confidence Score */}
              <Box sx={{ minWidth: 80, textAlign: 'center' }}>
                {getConfidenceIcon(selectedConfidence)}
                <Typography variant="caption">
                  {Math.round(selectedConfidence * 100)}%
                </Typography>
              </Box>

              {/* Detector Info */}
              <Typography variant="caption" color="text.secondary" sx={{ flex: 1 }}>
                {column.detector}
              </Typography>
            </Box>
          );
        })}

        {/* Summary */}
        <Box sx={{ mt: 2, pt: 2, borderTop: '1px solid #ddd' }}>
          <Typography variant="caption">
            ✓ Success Rate: {Math.round(
              (results.filter(r => r.confidence > 0.6).length / results.length) * 100
            )}% ({results.filter(r => r.confidence > 0.6).length}/{results.length} columns)
          </Typography>
        </Box>

        {/* Buttons */}
        <Box sx={{ display: 'flex', gap: 1, mt: 3, justifyContent: 'space-between' }}>
          <Button variant="outlined" onClick={onCancel}>
            Cancel
          </Button>
          <Button variant="contained" onClick={onConfirm}>
            Confirm & Continue →
          </Button>
        </Box>
      </CardContent>
    </Card>
  );
};
```

### Your API Call (Sally)

```jsx
// Main CsvUploadPanel.jsx
const handleUploadFile = async (file) => {
  // Validation
  if (!file.name.endsWith('.csv')) {
    setError('Invalid file format. Please upload a CSV file.');
    return;
  }

  if (file.size > 10 * 1024 * 1024) { // 10MB
    setError('File too large. Maximum size: 10 MB');
    return;
  }

  setIsLoading(true);
  setError(null);

  try {
    // Step 1: Parse CSV locally for preview
    const csvData = await new Promise((resolve, reject) => {
      Papa.parse(file, {
        header: true,
        skipEmptyLines: true,
        complete: (results) => resolve(results.data),
        error: (error) => reject(error)
      });
    });

    setCsvData(csvData);
    setPreview(csvData.slice(0, 10));

    // Step 2: Call backend API for type detection
    const formData = new FormData();
    formData.append('file', file);

    const response = await fetch(
      `/api/domains/${domainId}/detect-types`,
      {
        method: 'POST',
        body: formData,
        // DON'T set Content-Type - browser does it automatically
      }
    );

    if (!response.ok) {
      const errorData = await response.json().catch(() => ({}));
      throw new Error(errorData.message || 'Type detection failed');
    }

    const result = await response.json();
    setDetectionResults(result.data.columns);
    setStep('review'); // Show results

  } catch (err) {
    setError(err.message || 'An error occurred');
  } finally {
    setIsLoading(false);
  }
};
```
